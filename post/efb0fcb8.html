    此命令将启用运行 WSL 并安装 Linux 的 Ubuntu 发行版所需的功能。 （可以更改此默认发行版）。

    程序解题思路及代码

    1. sleep

      • 在完成该程序之前应先了解main函数中两个参数的概念，argc 为程序参数的个数，argv 为程序中的各个参数， 因此argc的最小值为1，即程序没有设置其他参数，可以此为判断条件判定程序是否有输入足够的参数。

      • atoi函数能将参数转化为整形数字，可使用atoi(argv[1])的形式获得睡眠时间

      • 调用user.h内sleep函数，完成程序，注意要有exit()，不然会报错

      • 程序源码：

        #include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"

int main(int argc, char *argv[]){

  if(argc < 2){//判断参数数量是否满足要求
    fprintf(2, "Usage:time outs\n");//参数数量不够，返回error
    exit(1);//退出
  }
 
  int time = atoi(argv[1]);//获取时间
  sleep(time);//调用sleep函数
  exit(0);//退出
}

    2. pingpong

      • 了解管道的概念(一端可写一端只读)及工作原理(半双工)

      • 注意管道读写顺序并且在读写完后应及时将端口关闭

      • 使用fork函数创建子进程，根据fork函数的返回值判断当前处于父进程还是子进程

      • 在父子进程同时进行读写操作时注意读写顺序，一个先读后写，一个先写后读，如果同时先读的话会造成死锁

      • 源码：

        #include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"

int main(int argc, char *argv[]){
    int p1[2];
    int p2[2];
    pipe(p1);//创建两个管道
    pipe(p2);
    int pid = fork();//创建子进程，返回子进程pid
    if(pid==0)//子进程
    {
        char son[1];
        close(p1[1]);//关闭管道写端
        read(p1[0], son, sizeof son);//读管道内数据，长度为一个字节
        close(p1[0]);//关闭管道读端

        printf("%d: received ping\n",getpid());//使用getpid()函数获取当前进程pid
        close(p2[0]);//关闭管道读端
        write(p2[1],"s",1);//将字符写入管道
        close(p2[1]);//关闭管道写端
        exit(0);//退出
    }
    else//farther
    {
        char far[1];
        close(p1[0]);//关闭管道读端
        write(p1[1],"f",1);//将字符写入管道
        close(p1[1]);//关闭管道写端

        close(p2[1]);//关闭管道写端
        read(p2[0], far, sizeof far);//读取管道内容，长度为一个字节
        close(p2[0]);//关闭管道读端
        printf("%d: received pong\n",getpid());//使用getpid()函数获取当前进程pid
         exit(0);//执行完毕，退出
    }
   
}

    3. primes

      • 新建整型数组，输入2到35的整数

      • 新建函数，参数为一个数组和数组长度

      • 函数首先输出第一个数组元素，之后新建管道并且创建子进程，并向管道内写入数组元素

      • 若处于子进程中，子进程不断读取管道内容，并且将获取的管道内容与数组第一个元素相除，若为0则不为质数，若不为0则将读取的数字置于当前input数组的首位，并将input指针位置向后移动，记录移动的次数为counter(此处采取的质数筛算法为欧拉筛法

      • 递归调用函数，此时数组参数须为input指针向前移动counter个位置，数组长度参数为counter

      • 若数组长度为1，则直接输出数组元素，此时该元素必为质数（因为已经过筛选）

      • 源码：

        #include "kernel/types.h"
#include "user/user.h"

void func(int *input, int num){//参数为一个数组和数组长度
    if(num == 1){//若数组长度为1，则直接输出数组元素（因为此时已经过筛选）
        printf("prime %d\n", *input);
        return;
    }
    int p[2],i;
    int prime = *input;//prime为数组首元素，必为质数
    int temp;
    printf("prime %d\n", prime);
    pipe(p);//新建管道
    if(fork() == 0){//子进程
        for(i = 0; i < num; i++){//将数组元素写入管道
            temp = *(input + i);
            write(p[1], (char *)(&temp), 4);
        }
        exit(0);
    }
    close(p[1]);//关闭管道写端
    if(fork() == 0){//子进程
        int counter = 0;//counter初始化
        char buffer[4];//
        while(read(p[0], buffer, 4) != 0){//循环，读至管道内无内容
            temp = *((int *)buffer);
            if(temp % prime != 0){//若从管道中读出的数字不能被数组首元素整除，则将其留下等待下轮筛选
                *input = temp;//置于当前数组指针input位置
                input += 1;//指针后移
                counter++;//记录后移次数，数组剩余长度
            }
        }
        func(input - counter, counter);//递归调用
        exit(0);
    }
    wait(0);//等待子进程执行
    wait(0);//等待子进程执行
}

int main(){
    int input[34];
    int i = 0;
    for(; i < 34; i++){//将2-35输入数组
        input[i] = i+2;
    }
    func(input, 34);调用函数
    exit(0);
}

      3. 阅读kalloc和kfree两个函数就可以知道，kmem.freelist是一个保存了当前空闲内存块的链表，因此只需要统计这个链表的长度再乘以PGSIZE就可以得到空闲内存。

        //kernel/kalloc.c
uint64
getfreemem(void)
{
  struct run *r;
  uint64 n = 0;

  acquire(&kmem.lock);
  r = kmem.freelist;
  while (r) {
    ++n;
    r = r->next;
  }
  release(&kmem.lock);

  return n * PGSIZE;
}

      4. 阅读procdump和相关代码就可以知道，XV6的进程结构体保存在proc[NPROC]数组当中。而proc->state字段保存了PCB的当前状态，有UNUSED、SLEEPING、RUNNABLE、RUNNING、ZOMBIE五种状态。因此只需要遍历这个数组，然后统计state不是UNUSED状态的就行了。

        //kernel/proc.c
uint64
getnproc(void)
{
  uint64 n = 0;
  struct proc *p;
  for (p = proc; p < &proc[NPROC]; ++p) {
    if (p->state != UNUSED) ++n;
  }
  return n;
}

Lab3：页表和进程的内存管理

3.1系统调用性能优化

这个 lab 的大概思路是，在创建进程时，就直接把进程的 pid 放入共享空间中，然后用户查询 pid 时，就不必通过 ecall 跳转到内核了，省去了保存现场等开销。

  • 首先需要为每一个进程多分配一个虚拟地址位于USYSCALL的页，然后这个页的开头保存一个usyscall结构体，结构体中存放这个进程的pid，在本实验中xv6已经帮我们定义好了：

    #ifdef LAB_PGTBL
#define USYSCALL (TRAPFRAME - PGSIZE)  // USYSCALl = 3FFFFFD000

struct usyscall {
    int pid;  // Process ID
};
#endif

3.2打印进程1的页表

这个思路就是递归遍历页表，碰到有效的就遍历进下一层页表，因为 xv6 的页表是多级的，所以是一个树的结构，那么本质上我们就是需要写一个通过 dfs 打印树的函数，在 kernel/vm.c 中把这个函数加入。

void 
vmprint(pagetable_t pagetable, uint dep){
  if(dep == 0)
    printf("page table %p\n", pagetable);
  for(int i = 0; i < 512; i++){
    pte_t pte = pagetable[i];
    if(pte & PTE_V){
      for(int j = 0; j < dep; j++)
        printf(".. ");
      uint64 child = PTE2PA(pte);
      printf("..%d: pte %p pa %p\n", i, pte, child);
      if(dep < 2)
        // 如果层数等于 2 就不需要继续递归了，因为这是叶子节点
        vmprint((pagetable_t) child, dep + 1);
    }
  } 
}

这个函数接收两个参数，要打印的页表（可以理解为要打印的树的根节点）和当前的深度，多出来一个深度是因为根据图片中的格式，我们需要根据当前的深度打印出不同数量的点。而且我们需要通过深度知道是否到达了叶子节点。

对于每个 pagetable，最多有 512 个节点，所以我们就依次遍历它们。如果发现这个页表是已分配的，也就是符合 pte & PTE_V == 1 的，我们就继续递归。

在打印的时候，我们先需要打印出 dep + 1 个 ..，然后再打印出 pte 和 pa。

这里指的 pte 指的是直接读取页表项的结果，而 pa 是去掉页表项中的标志位后得到的物理地址，我们通过这个物理地址可以找到下一层的页表项或是页帧。

注意可以这么 pte_t pte = pagetable[i]; 写是因为，pa 指向的实际上是这个子页表的第一个元素，而 pagetable[i] 和 *(pagetable + i) 是等价的，也就是去访问第 i 个页表。

这样这个 lab 中的主要部分就搞好了，下面我们可以去 kernel/exec.c 中的结尾插入以下代码：

if(p->pid == 1)
    vmprint(p->pagetable, 0);

因为 init 是系统创建的第一个进程，所以 init 的 pid 是 1，那么在创建 init 时，我们就会打印这个页表。

运行，发现报错，原因是vmprint函数未声明，因此需要在defs.h中加入声明

// vm.c
...
void            vmprint(pagetable_t pt,uint64);
...

成功运行

3.3跟踪页表的访问位

主要思路就是用户调用系统调用的时候，我们去查找页表，获得对应的PTE，然后检查PTE_A(需要自己定义)，然后决定是否在答案设置对应有效位。 在sysproc.c中完善sys_pgaccess():

#ifdef LAB_PGTBL
uint64 sys_pgaccess(void) {
    // lab pgtbl: your code here.
    
    uint64 buf;
    int number;
    uint64 ans;
    if (argaddr(0, &buf) < 0) return -1;
    if (argint(1, &number) < 0) return -1;
    if (argaddr(2, &ans) < 0) return -1;
    return pgaccess((void*)buf, number, (void*)ans);
}
#endif

然后去proc.c中加入pgaceess()的定义：

uint64 pgaccess(void *pg, int number, void *store) {
    struct proc *p = myproc();
    if (p == 0) {
        return 1;
    }
    pagetable_t pagetable = p->pagetable;
    int ans = 0;
    for (int i = 0; i < number; i++) {
        pte_t *pte;
        pte = walk(pagetable, ((uint64)pg) + (uint64)PGSIZE * i, 0);
        if (pte != 0 && ((*pte) & PTE_A)) {
            ans |= 1 << i;
            *pte ^= PTE_A;  // clear PTE_A
        }
    }
    // copyout
    return copyout(pagetable, (uint64)store, (char *)&ans, sizeof(int));
}

PTE_A 在 xv6 中还没被定义过，所以我们在 kernel/riscv.h 中定义一下：

#define PTE_A (1L << 6) // 左移六位是看上图决定的

由于sys_pgaccess函数调用了pgaccess，pgaccess内又调用了walk，有了上次的教训，这次依旧需要在defs.h中添加声明：(copyout已声明)

// proc.c
...
uint64          pgaccess(void *pg, int number, void *store);
...
// vm.c
...
pte_t *           walk(pagetable_t pagetable, uint64 va, int alloc);
...
